53InputState Keyboard::GetInputState(std::string_view keyid)
54{
55 auto enum_key = magic_enum::enum_cast<KeybordID>(keyid);
56 if (enum_key.has_value())
57 {
58 auto key = static_cast<size_t>(enum_key.value());
59 if (keybordData_[Trg::NOW][key] == 1 && keybordData_[Trg::OLD][key] != 1)
60 {
61 return InputState::PUSH;
62 }
63 if (keybordData_[Trg::NOW][key] != 1 && keybordData_[Trg::OLD][key] == 1)
64 {
65 return InputState::RELEASE;
66 }
67 if (keybordData_[Trg::NOW][key] == 1 && keybordData_[Trg::OLD][key] == 1)
68 {
69 return InputState::HOLD;
70 }
71 }
72 auto mouse_key = magic_enum::enum_cast<MouseInputID>(keyid);
73 if (mouse_key)
74 {
75 auto key = mouse_key.value();
76 if (mouseData_[key][Trg::NOW] && !mouseData_[key][Trg::OLD])
77 {
78 return InputState::PUSH;
79 }
80 if (!mouseData_[key][Trg::NOW] && mouseData_[key][Trg::OLD])
81 {
82 return InputState::RELEASE;
83 }
84 if (mouseData_[key][Trg::NOW] && mouseData_[key][Trg::OLD])
85 {
86 return InputState::HOLD;
87 }
88 }
