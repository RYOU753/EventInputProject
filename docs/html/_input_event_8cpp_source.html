84{
85 auto dir = GetAnalogVecEvent(id, id2).normalized();

88 if (isnan(r))

294{

317 eventTbl_[eventId].container[0] = keys;

321 {
322 if (!json["config"].find(it.key()).value()["Keyboard"].is_null())
323 {
324 eventTbl_[eventId].container[0] = json["config"].find(it.key()).value()["Keyboard"];

327 {
328 eventTbl_[eventId].container[1] = json["config"].find(it.key()).value()["Pad"];

371 auto now_sec = std::chrono::floor<std::chrono::seconds>(tz->to_local(now));
372 auto time = std::format("{:%Y%m%d_%H%M%S}", now_sec);
