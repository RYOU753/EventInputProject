192float XPadInput::GetMoveDirRot(short x, short y)
193{
194 auto dir = Vector2F(x, y);
195 auto kijun = Vector2F();
196
197 auto r = atan2(-(kijun.x - dir.x), -(kijun.y - dir.y));
198 if (r < 0)

200 r = r + PI2_F;

202
203 return floorf(r * 360 / PI2_F);
